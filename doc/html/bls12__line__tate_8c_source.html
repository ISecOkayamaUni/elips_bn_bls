31 static void bls12_EFp_ECD_return_lambda(EFp *ANS,Fp *lambda,EFp *P){
32  if(Fp_cmp_zero(&P->y)==0){
33  ANS->infinity=1;
34  return;
35  }
36 
37  EFp Tmp_P;
38  EFp_init(&Tmp_P);
39  EFp_set(&Tmp_P,P);
40  Fp tmp1,tmp2;
41  Fp_init(&tmp1);
42  Fp_init(&tmp2);
43 
44  Fp_mul_ui(&tmp1,&Tmp_P.y,2);
45  Fp_inv(&tmp1,&tmp1);
46  Fp_mul(&tmp2,&Tmp_P.x,&Tmp_P.x);
47  Fp_mul_ui(&tmp2,&tmp2,3);
48  Fp_mul(lambda,&tmp1,&tmp2);
49  Fp_mul(&tmp1,lambda,lambda);
50  Fp_mul_ui(&tmp2,&Tmp_P.x,2);
51  Fp_sub(&ANS->x,&tmp1,&tmp2);
52  Fp_sub(&tmp1,&Tmp_P.x,&ANS->x);
53  Fp_mul(&tmp2,lambda,&tmp1);
54  Fp_sub(&ANS->y,&tmp2,&Tmp_P.y);
55 
56  //clear
57  Fp_clear(&tmp1);
58  Fp_clear(&tmp2);
59  EFp_clear(&Tmp_P);
60 }

62 static void bls12_EFp_ECA_return_lambda(EFp *ANS,Fp *lambda,EFp *P1, EFp *P2){
63  if(P1->infinity==1){
64  EFp_set(ANS,P2);
65  return;
66  }else if(P2->infinity==1){
67  EFp_set(ANS,P1);
68  return;
69  }else if(Fp_cmp(&P1->x,&P2->x)==0){
70  if(Fp_cmp(&P1->y,&P2->y)!=0){
71  ANS->infinity=1;
72  return;
73  }else{
74  EFp_ECD(ANS,P1);
75  return;
76  }
77  }
78 
79  EFp Tmp_P1,Tmp_P2;
80  EFp_init(&Tmp_P1);
81  EFp_set(&Tmp_P1,P1);
82  EFp_init(&Tmp_P2);
83  EFp_set(&Tmp_P2,P2);
84  Fp tmp1,tmp2;
85  Fp_init(&tmp1);
86  Fp_init(&tmp2);
87 
88  Fp_sub(&tmp1,&Tmp_P2.x,&Tmp_P1.x);
89  Fp_inv(&tmp1,&tmp1);
90  Fp_sub(&tmp2,&Tmp_P2.y,&Tmp_P1.y);
91  Fp_mul(lambda,&tmp1,&tmp2);
92  Fp_mul(&tmp1,lambda,lambda);
93  Fp_sub(&tmp2,&tmp1,&Tmp_P1.x);
94  Fp_sub(&ANS->x,&tmp2,&Tmp_P2.x);
95  Fp_sub(&tmp1,&Tmp_P1.x,&ANS->x);
96  Fp_mul(&tmp2,lambda,&tmp1);
97  Fp_sub(&ANS->y,&tmp2,&Tmp_P1.y);
98 
99  //clear
100  Fp_clear(&tmp1);
101  Fp_clear(&tmp2);
102  EFp_clear(&Tmp_P1);
103  EFp_clear(&Tmp_P2);
104 }
