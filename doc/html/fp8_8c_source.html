98 void Fp8_mul(struct Fp8 *ANS,struct Fp8 *A, struct Fp8 *B){
99  struct Fp4 tmp1,tmp2;
100  Fp4_init(&tmp1);
101  Fp4_init(&tmp2);
102 
103  //set
104  Fp4_mul(&tmp2,&A->x1,&B->x1);//b*d
105  Fp4_add(&tmp1,&A->x0,&A->x1);//a+b
106  Fp4_add(&ANS->x1,&B->x0,&B->x1);//c+d
107  Fp4_mul(&ANS->x1,&tmp1,&ANS->x1);//(a+b)(c+d)
108  Fp4_mul(&tmp1,&A->x0,&B->x0);//a*c
109  //x0
110  Fp4_mul_basis(&ANS->x0,&tmp2);//b*d*v
111  Fp4_add(&ANS->x0,&ANS->x0,&tmp1);//a*c+b*d*v
112  //x1
113  Fp4_sub(&ANS->x1,&ANS->x1,&tmp1);
114  Fp4_sub(&ANS->x1,&ANS->x1,&tmp2);
115 
116  //clear
117  Fp4_clear(&tmp1);
118  Fp4_clear(&tmp2);
119 }

192 void Fp8_sqrt(struct Fp8 *ANS,struct Fp8 *A){
193  struct Fp8 n,y,x,b,t,tmp_Fp4;
194  Fp8_init(&n);
195  Fp8_init(&y);
196  Fp8_init(&x);
197  Fp8_init(&b);
198  Fp8_init(&t);
199  Fp8_init(&tmp_Fp4);
200  Fp8_set(&n,A);
201

203  mpz_init(tmp_mpz);
204  mpz_init(q);
205  mpz_init(e);
206  mpz_init(r);
207  mpz_init(set_1);
208  mpz_init(set_2);
209  mpz_set_ui(set_1,1);
210  mpz_set_ui(set_2,2);
211 
212  gmp_randstate_t state;
213  gmp_randinit_default(state);
214  gmp_randseed_ui(state,(unsigned long)time(NULL));
215 
216  Fp8_set_random(&n,state);
217  while(Fp8_legendre(&n)!=-1){
218  Fp8_set_random(&n,state);
219  }
220 
221  mpz_pow_ui(q,curve_parameters.prime,12);
222  mpz_sub_ui(q,q,1);
223  mpz_set_ui(e,0);
224  while(mpz_odd_p(q)==0){
225  mpz_add_ui(e,e,1);
226  mpz_div_ui(q,q,2);
227  }
228  Fp8_pow(&y,&n,q);
229 
230  mpz_set(r,e);
231 
232  mpz_sub_ui(tmp_mpz,q,1);
233  mpz_div_ui(tmp_mpz,tmp_mpz,2);
234 
235  Fp8_pow(&x,A,tmp_mpz);
236  Fp8_pow(&tmp_Fp4,&x,set_2);
237  Fp8_mul(&b,&tmp_Fp4,A);
238  Fp8_mul(&x,&x,A);
239 
240  int m;
241 
242  while(Fp8_cmp_mpz(&b,set_1)==1){
243  m=-1;
244  Fp8_set(&tmp_Fp4,&b);
245  while(Fp8_cmp_mpz(&tmp_Fp4,set_1)==1){
246  m++;
247  mpz_pow_ui(tmp_mpz,set_2,m);
248  Fp8_pow(&tmp_Fp4,&b,tmp_mpz);
249  }
250  mpz_sub_ui(tmp_mpz,r,m);
251  mpz_sub_ui(tmp_mpz,tmp_mpz,1);
252  mpz_powm(tmp_mpz,set_2,tmp_mpz,curve_parameters.prime);
253  // gmp_printf("%Zd,%Zd,%d\n",tmp_mpz,r,m);
254  Fp8_pow(&t,&y,tmp_mpz);
255  Fp8_pow(&y,&t,set_2);
256  // gmp_printf("%Zd,%Zd,\n",y.x0.x0.x0,y.x0.x1.x0);
257  mpz_set_ui(r,m);
258  Fp8_mul(&x,&x,&t);
259  Fp8_mul(&b,&b,&y);
260  }
261 
262  Fp8_set(ANS,&x);
263 
264  Fp8_clear(&n);
265  Fp8_clear(&y);
266  Fp8_clear(&x);
267  Fp8_clear(&b);

270  mpz_clear(tmp_mpz);
271  mpz_clear(q);
272  mpz_clear(e);
273  mpz_clear(r);
274  mpz_clear(set_1);
275 }
