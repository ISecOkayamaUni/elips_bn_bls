85 void Fp6_mul(Fp6 *ANS,Fp6 *A,Fp6 *B){
86  Fp2 tmp00,tmp11,tmp22,tmp,t0,t1,t2;
87  Fp2_init(&tmp00);
88  Fp2_init(&tmp11);
89  Fp2_init(&tmp22);
90  Fp2_init(&tmp);
91  Fp2_init(&t0);
92  Fp2_init(&t1);
93  Fp2_init(&t2);
94 
95  //set
96  Fp2_mul(&tmp00,&A->x0,&B->x0);//x0*y0
97  Fp2_mul(&tmp11,&A->x1,&B->x1);//x1*y1
98  Fp2_mul(&tmp22,&A->x2,&B->x2);//x2*y2
99 
100  Fp2_add(&t0,&A->x0,&A->x1);//x0+x1
101  Fp2_add(&tmp,&B->x0,&B->x1);//y0+y1
102  Fp2_mul(&t0,&t0,&tmp);//(x0+x1)(y0+y1)
103 
104  Fp2_add(&t1,&A->x1,&A->x2);//x1+x2
105  Fp2_add(&tmp,&B->x1,&B->x2);//y1+y2
106  Fp2_mul(&t1,&t1,&tmp);//(x1+x2)(y1+y2)
107 
108  Fp2_add(&t2,&B->x0,&B->x2);//y2+y0
109  Fp2_add(&tmp,&A->x0,&A->x2);//x2+x0
110  Fp2_mul(&t2,&t2,&tmp);//(x2+x0)(y2+y0)
111  //x0
112  Fp2_sub(&t1,&t1,&tmp11);
113  Fp2_sub(&t1,&t1,&tmp22);//(x1+x2)(y1+y2)-x1y1-x2y2
114  Fp2_mul_basis(&tmp,&t1);
115  Fp2_add(&ANS->x0,&tmp00,&tmp);
116  //x1
117  Fp2_sub(&t0,&t0,&tmp00);
118  Fp2_sub(&t0,&t0,&tmp11);
119  Fp2_mul_basis(&tmp,&tmp22);
120  Fp2_add(&ANS->x1,&tmp,&t0);
121  //x2
122  Fp2_sub(&t2,&t2,&tmp00);
123  Fp2_sub(&t2,&t2,&tmp22);
124  Fp2_add(&ANS->x2,&tmp11,&t2);
125 
126  //clear
127  Fp2_clear(&tmp00);
128  Fp2_clear(&tmp11);
129  Fp2_clear(&tmp22);
130  Fp2_clear(&tmp);
131  Fp2_clear(&t0);
132  Fp2_clear(&t1);
133  Fp2_clear(&t2);
134 }

163 void Fp6_sqr(Fp6 *ANS,Fp6 *A){
164  Fp2 tmp00,tmp12_2,tmp01_2,tmp22,tmp;
165  Fp2_init(&tmp00);
166  Fp2_init(&tmp22);
167  Fp2_init(&tmp12_2);
168  Fp2_init(&tmp01_2);
169  Fp2_init(&tmp);
170 
171  Fp2_sqr(&tmp00,&A->x0); //x0^2
172  Fp2_sqr(&tmp22,&A->x2); //x2^2
173  Fp2_add(&tmp,&A->x1,&A->x1); //2x1
174  Fp2_mul(&tmp12_2,&tmp,&A->x2); //2x1x2
175  Fp2_mul(&tmp01_2,&A->x0,&tmp); //2x0x1
176  Fp2_add(&tmp,&A->x0,&A->x1); //x0+x1+x2
177  Fp2_add(&tmp,&tmp,&A->x2);
178 
179  //x0
180  Fp2_mul_basis(&ANS->x0,&tmp12_2);
181  Fp2_add(&ANS->x0,&ANS->x0,&tmp00);
182  //x1
183  Fp2_mul_basis(&ANS->x1,&tmp22);
184  Fp2_add(&ANS->x1,&ANS->x1,&tmp01_2);
185  //x2
186  Fp2_sqr(&ANS->x2,&tmp);
187  Fp2_add(&tmp,&tmp00,&tmp22);
188  Fp2_add(&tmp,&tmp,&tmp12_2);
189  Fp2_add(&tmp,&tmp,&tmp01_2);
190  Fp2_sub(&ANS->x2,&ANS->x2,&tmp);
191 
192  Fp2_clear(&tmp00);
193  Fp2_clear(&tmp22);
194  Fp2_clear(&tmp12_2);
195  Fp2_clear(&tmp01_2);
196  Fp2_clear(&tmp);
197 }

235 void Fp6_inv(Fp6 *ANS,Fp6 *A){
236  Fp6 frob1,frob2,tmp1,tmp2;
237  Fp6_init(&frob1);
238  Fp6_init(&frob2);
239  Fp6_init(&tmp1);
240  Fp6_init(&tmp2);
241 
242  Fp2_set(&frob1.x0,&A->x0);
243  Fp2_mul_mpz(&frob1.x1,&A->x1,epsilon1);
244  Fp2_mul_mpz(&frob1.x2,&A->x2,epsilon2);
245 
246  Fp2_set(&frob2.x0,&A->x0);
247  Fp2_mul_mpz(&frob2.x1,&A->x1,epsilon2);
248  Fp2_mul_mpz(&frob2.x2,&A->x2,epsilon1);
249 
250  Fp6_mul(&tmp1,&frob1,&frob2);
251  Fp6_mul(&tmp2,&tmp1,A);
252  Fp2_inv(&tmp2.x0,&tmp2.x0);
253  Fp2_mul(&ANS->x0,&tmp1.x0,&tmp2.x0);
254  Fp2_mul(&ANS->x1,&tmp1.x1,&tmp2.x0);
255  Fp2_mul(&ANS->x2,&tmp1.x2,&tmp2.x0);
256 
257  Fp6_clear(&frob1);
258  Fp6_clear(&frob2);
259  Fp6_clear(&tmp1);
260  Fp6_clear(&tmp2);
261 }

344 void Fp6_pow(Fp6 *ANS,Fp6 *A,mpz_t scalar){
345  int i,length;

347  char binary[length];
348  mpz_get_str(binary,2,scalar);
349  Fp6 tmp;

354  Fp6_sqr(&tmp,&tmp);
